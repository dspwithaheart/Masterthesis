\newglossaryentry{Inkrementalgeber}
{
	name = Inkrementalgeber,
	description= {Ein Inkrementalgeber (auch \textbf{Drehgeber} oder \textbf{Encoder}) liefert eine bestimmte Anzahl an Impulsen pro Umdrehung seiner Welle. Für die Drehrichtungserfassung werden zwei um 90° versetzte Ausgangssignale gesendet (A- und B -Spur). Durch Zählen der Impulse lässt sich ein Drehwinkel ermitteln. Einsatz finden Inkrementalgeber häufig in regelungstechnischen Aufgaben.}
}
